\doxysubsection{Detailed Description}
{\itshape LIS-\/770i configuration} 


\begin{DoxyItemize}
\item sensor config is three bytes\+: {\ttfamily binning}, {\ttfamily gain}, and {\ttfamily active\+\_\+rows}
\begin{DoxyItemize}
\item {\ttfamily binning}\+: control LIS-\/770i pixel pitch
\begin{DoxyItemize}
\item 0x00 (off) -- use 7.\+8µm pixel pitch
\item 0x01 (on) -- use 15.\+6µm pixel pitch
\end{DoxyItemize}
\item {\ttfamily gain}\+: there are four gain settings in the LIS-\/770i
\begin{DoxyItemize}
\item 0x01 (1x)
\item 0x25 (2.\+5x) (0x25 is 37 in decimal)
\item 0x04 (4x)
\item 0x05 (5x)
\end{DoxyItemize}
\item {\ttfamily active\+\_\+rows}\+: there are five rows in the LIS-\/770i
\begin{DoxyItemize}
\item set the bit to turn the row on
\item clear the bit to turn the row off
\item the 5 least significant bits represent the 5 rows
\item bits \mbox{[}8..0\mbox{]}\+: xxx54321
\begin{DoxyItemize}
\item {\ttfamily 1} to {\ttfamily 5} are the row numbers
\item {\ttfamily x} is don\textquotesingle{}t care
\end{DoxyItemize}
\end{DoxyItemize}
